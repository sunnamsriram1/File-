import pyttsx3

vtf = pyttsx3.init()
rate = vtf.getProperty('rate')
print(rate)                               # getting details of current speaking rate

def vtf_speak(text):
    
     print('Speaking.......')
     print(' ')
     vtf.say(text)
     vtf.runAndWait()
